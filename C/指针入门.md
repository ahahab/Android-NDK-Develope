##指针与二维数组、函数指针,指针与二维数组、函数指针

```c
//函数指针
//用随机数生成一个数组，写一个函数查找最小的值，并返回最小数的地址，在主函数中打印出来
int* getMinPointer(int ids[], int len){
	int i = 0; 	
	int* p = &ids[0];
	for (; i < len; i++){
		if (ids[i] < *p){			
			p = &ids[i];
		}
	}
	return p;
}

void main(){
	int ids[10];
	int i = 0;
	//初始化随机数发生器，设置种子，种子不一样，随机数才不一样
	//当前时间作为种子 有符号 int -xx - > +xx
	srand((unsigned)time(NULL));
	for (; i < 10; i++){
		//100范围内
		ids[i] = rand() % 100;
		printf("%d\n", ids[i]);
	}

	int* p = getMinPointer(ids, sizeof(ids) / sizeof(int));
	printf("%#x,%d\n",p,*p);
	getchar();
}

```

3.结构体函数指针成员
```c
typedef struct Girl{
	char *name;
	int age;
	//函数指针
	void(*sayHi)(char*);
}Girl;

//Girl结构体指针取别名GirlP
typedef Girl* GirlP;

void sayHi(char* text){
	MessageBoxA(0, text, "title", 0);
}

//改名
void rename(GirlP gp1){
	gp1->name = "Lily";
}

void main(){
	Girl g1 = { "Lucy", 18, sayHi };
	GirlP gp1 = &g1;
	gp1->sayHi("Byebye!");
	//传递指针，改名
	rename(gp1);

	getchar();
}
```
